Konwerter jednostek

window.onload = function() { const unit = prompt("Wybierz jednostkę do przeliczenia (km/miles):"); const value = parseFloat(prompt("Podaj wartość do przeliczenia:")); let result; if (unit === 'km') { result = value * 0.621371 + ' mil'; } else if (unit === 'miles') { result = value / 0.621371 + ' km'; } else { result = "Nieprawidłowa jednostka!"; } document.getElementById('result').innerText = `Wynik: ${result}`; } body { font-family: Verdana, sans-serif; background-color: #cae7ff; text-align: center; padding: 50px; } .container { width: 600px; margin: 0 auto; padding: 24px; border: 2px solid red; background-color: white; } h1 { color: #006ec9; } #result { margin-top: 20px; font-size: 30px; color: red; }
