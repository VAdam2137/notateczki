Konwerter jednostek

window.onload = function() { let unit; let value; let result; while (true) { unit = prompt("Wybierz jednostkę do przeliczenia (km/miles):"); if (unit === 'km' || unit === 'miles') { break; } else { alert("Nieprawidłowa jednostka! Wybierz 'km' lub 'miles'."); } } while (true) { value = prompt("Podaj wartość do przeliczenia:"); if (!isNaN(value) && value.trim() !== "") { value = parseFloat(value); break; } else { alert("Nieprawidłowa wartość! Podaj liczbę."); } } if (unit === 'km') { result = (value * 0.621371).toFixed(2) + ' mil'; } else if (unit === 'miles') { result = (value / 0.621371).toFixed(2) + ' km'; } document.getElementById('result').innerText = `Wynik: ${result}`; } body { font-family: Verdana, sans-serif; background-color: #cae7ff; text-align: center; padding: 50px; } .container { width: 600px; margin: 0 auto; padding: 24px; border: 2px solid red; background-color: white; border-radius: 30px; } h1 { color: #006ec9; } #result { margin-top: 20px; font-size: 30px; color: red; }
